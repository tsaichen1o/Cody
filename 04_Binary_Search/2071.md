    bool canComplete(int k, vector<int>& tasks, vector<int>& workers, int pills, int strength) {
        multiset<int> s(workers.end() - k, workers.end());  // å–æœ€å¼· k å€‹å·¥äºº
        
        int p = pills;

        for (int i = k - 1; i >= 0; --i) {  // å¾æœ€é›£çš„ä»»å‹™é–‹å§‹åˆ†é…
            int task = tasks[i];

        return true;
    }

    int maxTaskAssign(vector<int>& tasks, vector<int>& workers, int pills, int strength) {
        sort(tasks.begin(), tasks.end());
        sort(workers.begin(), workers.end());

        int l = 0, r = min((int)tasks.size(), (int)workers.size());
        int res = 0;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (canComplete(mid, tasks, workers, pills, strength)) {
                res = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return res;
    }
};
